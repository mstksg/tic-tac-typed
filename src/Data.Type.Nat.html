$(singletons [d|
  data N = Z | S N
    deriving (Show, Eq, Ord)


  addN :: N -> N -> N
  addN Z     m = m
  addN (S n) m = S (addN n m)
  |])

data LT :: N -> N -> Type where
    LTZ :: LT 'Z n
    LTS :: LT n m -> LT ('S n) ('S m)
